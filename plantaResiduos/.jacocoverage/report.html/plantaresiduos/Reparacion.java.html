import java.util.Calendar;

    private Calendar fecha;
    private String desc;

    /**
     * 
     * @param id - id de la reparación.
     * @param fecha - fecha de ésta.
     * @param desc - descripción de la misma.
     */

        this.id = id;
        this.fecha = fecha;
        this.desc = desc;
    }
